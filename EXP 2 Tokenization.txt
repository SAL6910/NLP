import nltk
from nltk.tokenize import word_tokenize
from nltk.tokenize import sent_tokenize

str="Hello how are you ? How you doing"
print(type(str))
string=str.lower()
stok=sent_tokenize(string)
wtok=word_tokenize(string)
print(type(stok))
print(type(wtok))
print(stok)
print(wtok)


Output:

<class 'str'>
<class 'list'>
<class 'list'>
['hello how are you ?', 'how you doing']
['hello', 'how', 'are', 'you', '?', 'how', 'you', 'doing']
